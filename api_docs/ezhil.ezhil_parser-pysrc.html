537              [l, c] = tok_num.get_line_col() 
538              val = Number( tok.val , l, c, self.debug ) 
539          elif tok.kind == EzhilToken.LOGICAL_NOT: 
540              tok_not = self.dequeue() 
541              [l, c] = tok_not.get_line_col() 
542              val = UnaryExpr( self.expr(), tok_not , l, c, self.debug ) 
543              self.dbg_msg("completed parsing unary expression"+str(val)) 
544          elif tok.kind ==  EzhilToken.ID: 
545              tok_id = self.dequeue() 
546              [l, c] = tok_id.get_line_col() 
547              val = Identifier( tok.val , l, c, self.debug ) 
548              ptok = self.peek() 
549              self.dbg_msg("factor: "+str(ptok) + " / "+str(tok) ) 
550              if ( ptok.kind ==  EzhilToken.LPAREN ): 
551                  ## function call 
552                  [l, c] = ptok.get_line_col() 
553                  vallist = self.valuelist() 
554                  val=ExprCall( val, vallist, l, c, self.debug ) 
555              elif ( ptok.kind ==  EzhilToken.LSQRBRACE ): 
556                  ## indexing a array type variable or ID 
557                  val=None 
558                  raise ParseException("arrays not implemented"+str(ptok)); 
559          elif tok.kind ==  EzhilToken.STRING : 
560              str_tok = self.dequeue() 
561              [l,c] = str_tok.get_line_col() 
562              val = String( tok.val , l, c, self.debug ) 
563          elif tok.kind in EzhilToken.ADDSUB: 
564              unop = self.dequeue(); 
565              [l, c] = unop.get_line_col() 
566              val=Expr(Number(0),unop,self.term(),l,c,self.debug); 
567          elif tok.kind == EzhilToken.LSQRBRACE: 
568              # creating a list/array expression 
569              list_start = self.dequeue(); 
570              val = Array() 
571              while( True ): 
572                  exprval = self.expr() 
573                  val.append( exprval  ) 
574                  if self.debug : print(self.peek().__class__,self.peek()) 
575                  if ( self.peek().kind == EzhilToken.RSQRBRACE ): 
576                      break 
577                  else: 
578                      assert( self.peek().kind == EzhilToken.COMMA) 
579                      self.dequeue() 
580              assert( self.peek().kind == EzhilToken.RSQRBRACE ) 
581              list_end = self.dequeue() 
582          else: 
583              raise ParseException("Expected Number, found something "+str(tok)) 
584           
585          self.dbg_msg( "factor-returning: "+str(val) ) 
586          return val
